def read_until(rd: socket.socket):
    # leftover data
    buf = b''

    while data := rd.recv(1024):
        # combine leftover data with incoming data
        cuts = (buf + data).split(LF)

        # leftover data
        buf = cuts[-1]

        for l in cuts[:-1]:
            # connvert to string
            yield l.decode()

    # connvert to string
    yield buf.decode()
